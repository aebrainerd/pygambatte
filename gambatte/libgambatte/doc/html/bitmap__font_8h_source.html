59 template<class RandomAccessIterator, class Fill>
60 void print(RandomAccessIterator dest, std::ptrdiff_t const pitch, Fill fill, char const *chars) {
61  while (int const character = *chars++) {
62  RandomAccessIterator dst = dest;
63  unsigned char const *s = font[character];
64  unsigned const width = *s >> 4;
65  unsigned h = *s++ & 0xF;
66 
67  while (h--) {
68  RandomAccessIterator d = dst;
69  unsigned line = *s++;
70 
71  if (width > 8)
72  line |= *s++ << 8;
73 
74  while (line) {
75  if (line & 1)
76  fill(d, pitch);
77 
78  line >>= 1;
79  ++d;
80  }
81 
82  dst += pitch;
83  }
84 
85  dest += width;
86  }
87 }

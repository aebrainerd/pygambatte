80  unsigned ff_read(unsigned p, unsigned cc) {
81  uint8_t v = p < 0x80 ? nontrivial_ff_read(p, cc) : ioamhram_[p + 0x100];
82  if(__builtin_expect(dbg->ioamhram[0x100 + p] & 1, 0))
83  dbg->read(2, 0x100 + p, v, false);
84  if(__builtin_expect(dbg->bus[0xFF00 + p] & 1, 0))
85  dbg->read(0, 0xFF00 + p, v, false);

89  inline uint8_t do_read_trap(const uint8_t* addr, std::pair<unsigned char*, size_t> area, unsigned clazz,
90  const uint8_t* dbgflags, std::map<unsigned, uint8_t>& cheats, uint8_t v, uint8_t mask, bool exec)
91  {
92  if(addr >= area.first && addr < area.first + area.second) {
93  if(__builtin_expect(dbgflags[addr - area.first] & mask, 0)) {
94  if(dbgflags[addr - area.first] & (mask & 77))
95  dbg->read(clazz, addr - area.first, v, exec);
96  if(__builtin_expect(dbgflags[addr - area.first] & 8, 0)) {
97  auto itr = cheats.find(addr - area.first);
98  if(itr != cheats.end()) v = itr->second;
99  }
100  }
101  }
102  return v;
103  }

105  inline void do_write_trap(const uint8_t* addr, std::pair<unsigned char*, size_t> area, unsigned clazz,
106  const uint8_t* dbgflags, uint8_t v)
107  {
108  if(addr >= area.first && addr < area.first + area.second)
109  if(__builtin_expect(dbgflags[addr - area.first] & 0x22, 0))
110  dbg->write(clazz, addr - area.first, v);
111  }

113  unsigned read(unsigned p, unsigned cc, bool exec) {
114  uint8_t mask = exec ? 0x4C : 0x19;
115  const unsigned char* memblock = cart_.rmem(p >> 12);
116  uint8_t v = memblock ? memblock[p] : nontrivial_read(p, cc, exec);
117  uint8_t v2 = v;
118  if(memblock) {
119  if(p >= 0xFE00) { //IOAMHRAM.
120  if(__builtin_expect(dbg->ioamhram[p - 0xFE00] & mask, 0))
121  dbg->read(2, 0x100 + p, v, exec);
122  } else {
123  const uint8_t* addr = memblock + p;
124  static void* targets[8] = {&&cart, &&cart, &&cart, &&cart, &&out, &&sram, &&wram,
125  &&wram};
126  goto *targets[p >> 13];
127 wram:
128  v2 = do_read_trap(addr, cart_.getWorkRam(), 1, dbg->wram, dbg->wramcheat, v, mask, exec);
129  goto out;
130 sram:
131  v2 = do_read_trap(addr, cart_.getSaveRam(), 4, dbg->sram, dbg->sramcheat, v, mask, exec);
132  goto out;
133 cart:
134  v2 = do_read_trap(addr, cart_.getCartRom(), 3, dbg->cart, dbg->cartcheat, v, mask, exec);
135  goto out;
136  }
137 out: ;
138  }
139  if(__builtin_expect(dbg->bus[p] & mask, 0))
140  dbg->read(0, p, v, exec);
141  return v2;
142  }

144  void write(unsigned p, unsigned data, unsigned cc) {
145  if(__builtin_expect(dbg->bus[p] & 0x22, 0))
146  dbg->write(0, p, data);
147  unsigned char* memblock = cart_.wmem(p >> 12);
148  if(memblock) {
149  if(p >= 0xFE00) //IOAMHRAM.
150  if(__builtin_expect(dbg->ioamhram[p - 0xFE00] & 2, 0))
151  dbg->write(2, 0x100 + p, data);
152  uint8_t* addr = memblock + p;
153  do_write_trap(addr, cart_.getWorkRam(), 1, dbg->wram, data);
154  do_write_trap(addr, cart_.getSaveRam(), 4, dbg->sram, data);
155  do_write_trap(addr, cart_.getCartRom(), 3, dbg->cart, data);
156  }
157  if (memblock) {
158  memblock[p] = data;
159  } else
160  nontrivial_write(p, data, cc);
161  }

168  if (p - 0x80u < 0x7Fu) {
169  ioamhram_[p + 0x100] = data;
170  } else
171  nontrivial_ff_write(p, data, cc);
172  }

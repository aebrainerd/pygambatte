40 // Keeps track of minimum value identified by id as values change.
41 // Higher ids prioritized (as min value) if values are equal. Can easily be reversed by swapping < for <=.
42 // Higher ids can be faster to change when the number of ids isn't a power of 2.
43 // Thus the ones that change more frequently should have higher ids if priority allows it.
44 template<int ids>
45 class MinKeeper {

126 template<int ids>
127 MinKeeper<ids>::MinKeeper(unsigned const initValue) {
128  std::fill(values_, values_ + ids, initValue);
129 
130  for (int i = 0; i < Num<levels-1>::r; ++i) {
131  a_[Sum<levels-1>::r + i] = i * 2 + 1 == ids || values_[i * 2] < values_[i * 2 + 1]
132  ? i * 2
133  : i * 2 + 1;
134  }
135 
136  int n = Num<levels-1>::r;
137  int off = Sum<levels-1>::r;
138 
139  while (off) {
140  int const pn = (n + 1) >> 1;
141  int const poff = off - pn;
142 
143  for (int i = 0; i < pn; ++i) {
144  a_[poff + i] =
145  i * 2 + 1 == n || values_[a_[off + i * 2]] < values_[a_[off + i * 2 + 1]]
146  ? a_[off + i * 2 ]
147  : a_[off + i * 2 + 1];
148  }
149 
150  off = poff;
151  n = pn;
152  }
153 
154  minValue_ = values_[a_[0]];
155 }

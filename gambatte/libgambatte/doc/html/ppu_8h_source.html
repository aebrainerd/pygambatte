38 class PPUFrameBuf {
39 public:
40  PPUFrameBuf() : fbline_(nullfbline()), pitch_(160) { memset(buf_, 0, sizeof(buf_)); }
41  uint_least32_t * fb() const { return buf_; }
42  uint_least32_t * fbline() const { return fbline_; }
43  std::ptrdiff_t pitch() const { return pitch_; }
44  void setFbline(unsigned ly) { fbline_ = buf_ ? buf_ + std::ptrdiff_t(ly) * pitch_ : nullfbline(); }
45  void blit(uint_least32_t *const buf, const int pitch) const {
46  for(unsigned i = 0; i < 144; i++)
47  memcpy(buf + i * static_cast<signed>(pitch), buf_ + i * 160, 160 * sizeof(buf[0]));
48  }
49  void loadOrSave(loadsave& state) {
50  state(buf_, 160*144);
51  state(pitch_);
52  bool var = (fbline_ != nullfbline());
53  state(var);
54  if(var) {
55  unsigned x = fbline_ - buf_;
56  state(x);
57  fbline_ = buf_ + x;
58  } else
59  fbline_ = nullfbline();
60  }
61 
62 private:
63  mutable uint_least32_t buf_[160*144];
64  uint_least32_t *fbline_;
65  int pitch_;
66 
67  static uint_least32_t * nullfbline() { static uint_least32_t nullfbline_[160]; return nullfbline_; }
68 };

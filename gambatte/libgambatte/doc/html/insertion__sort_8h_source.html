25 void insertionSort(T *const start, T *const end, Less less) {
26  if (start >= end)
27  return;
28 
29  T *a = start;
30 
31  while (++a < end) {
32  T const e = *a;

34 
35  while (b != start && less(e, *(b - 1))) {
36  *b = *(b - 1);
37  b = b - 1;
38  }
39 
40  *b = e;
41  }
42 }
